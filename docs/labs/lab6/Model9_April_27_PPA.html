# calculating margin of error (MOE / Estimate * 100 = coefficient of variation)
pa_county_data <- pa_county_data %>%
  mutate(
    MEDHHINC_MOE   = (MEDHHINCM    / MEDHHINCE)   * 100,  # CV for median household income
    WHITE_MOE      = (WHITEM       / WHITEE)       * 100,  # CV for white population
    TOTAL_OCC_MOE  = (TOTAL_OCCM   / TOTAL_OCCE)  * 100,  # CV for total occupied units
    OWNER_OCC_MOE  = (OWNER_OCCM   / OWNER_OCCE)  * 100,  # CV for owner occupied units
    RENTER_OCC_MOE = (RENTER_OCCM  / RENTER_OCCE) * 100,  # CV for renter occupied units
    TOTAL_HSG_MOE  = (TOTAL_HSGM   / TOTAL_HSGE)  * 100,  # CV for total housing units
    VACANT_MOE     = (VACANTM      / VACANTE)      * 100,  # CV for vacant units
    OWN_TOT_MOE    = (OWN_TOTM     / OWN_TOTE)    * 100,  # CV for owner move-in universe
    RNT_TOT_MOE    = (RNT_TOTM     / RNT_TOTE)    * 100,  # CV for renter move-in universe
    BUILT_TOT_MOE  = (BUILT_TOTM   / BUILT_TOTE)  * 100,  # CV for year built universe
    TPOP_MOE       = (TPOPM        / TPOPE)        * 100   # CV for total population
  )

# categorizing reliability based on CV thresholds
filter_pa_category <- pa_county_data %>%
  mutate(
    reliability = case_when(
      # Unreliable: estimate is zero or missing (CV is NA or infinite)
      is.na(MEDHHINC_MOE)   | is.infinite(MEDHHINC_MOE)   |
      is.na(TOTAL_HSG_MOE)  | is.infinite(TOTAL_HSG_MOE)  |
      is.na(VACANT_MOE)     | is.infinite(VACANT_MOE)     |
      is.na(TPOP_MOE)       | is.infinite(TPOP_MOE)       |
      is.na(OWNER_OCC_MOE)  | is.infinite(OWNER_OCC_MOE)  |
      is.na(RENTER_OCC_MOE) | is.infinite(RENTER_OCC_MOE) ~ "Unreliable",

      # Use with caution: CV exceeds 40% on any core variable
      MEDHHINC_MOE   > 40 |
      TOTAL_HSG_MOE  > 40 |
      VACANT_MOE     > 40 |
      TPOP_MOE       > 40 |
      OWNER_OCC_MOE  > 40 |
      RENTER_OCC_MOE > 40 ~ "Use with Caution",

      TRUE ~ "Reliable"
    )
  ) %>%
  drop_na()

    OWN_BURDEN_CV  = (OWN_BURDEN_TOTM / OWN_BURDEN_TOTE) * 100,
    RNT_BURDEN_CV  = (RNT_BURDEN_TOTM / RNT_BURDEN_TOTE) * 100,
    SNAP_CV        = (SNAP_TOTM     / SNAP_TOTE)       * 100
  )


# ── Reliability classification (CV threshold: 40%) ───────────────────────────
philly_tract_filtered <- philly_tract_data %>%
  mutate(
    reliability = case_when(
      # Unreliable: estimate is zero/missing → CV is NA or infinite
      is.na(UNEMP_CV)      | is.infinite(UNEMP_CV)      |
      is.na(EDUC_CV)       | is.infinite(EDUC_CV)       |
      is.na(POV_CV)        | is.infinite(POV_CV)        |
      is.na(OWN_BURDEN_CV) | is.infinite(OWN_BURDEN_CV) |
      is.na(RNT_BURDEN_CV) | is.infinite(RNT_BURDEN_CV) |
      is.na(SNAP_CV)       | is.infinite(SNAP_CV)
      ~ "Unreliable",

      # Use with caution: any core denominator CV exceeds 40%
      UNEMP_CV      > 40 |
      EDUC_CV       > 40 |
      POV_CV        > 40 |
      OWN_BURDEN_CV > 40 |
      RNT_BURDEN_CV > 40 |
      SNAP_CV       > 40
      ~ "Use with Caution",

      TRUE ~ "Reliable"
    )
  ) %>%
  drop_na()


# ── Derived rates ─────────────────────────────────────────────────────────────
final_tract_dataset <- philly_tract_filtered %>%
  mutate(
    # Unemployment rate: unemployed / civilian labor force
    UNEMP_RATE     = (UNEMPLOYEDE  / LBR_FORCEE)       * 100,

    # Bachelor's degree or higher: sum of bach + grad degrees / pop 25+
    BACH_PLUS      = BACHE + MASTERSE + PROFE + DOCTE,
    BACH_RATE      = (BACH_PLUS    / EDUC_TOTE)        * 100,

    # Poverty rate: population below poverty / total poverty universe
    POV_RATE       = (POV_BELOWE   / POV_TOTE)         * 100,

    # Cost-burdened owners (≥30% income on housing costs)
    OWN_BURDENED   = OWN_30_34E + OWN_35_39E + OWN_40_49E + OWN_50PE,
    OWN_BURDEN_RATE= (OWN_BURDENED / OWN_BURDEN_TOTE)  * 100,

    # Cost-burdened renters (≥30% income on gross rent)
    RNT_BURDENED   = RNT_30_34E + RNT_35_39E + RNT_40_49E + RNT_50PE,
    RNT_BURDEN_RATE= (RNT_BURDENED / RNT_BURDEN_TOTE)  * 100,

    # Overall cost-burden rate (owners + renters combined)
    TOTAL_BURDENED = OWN_BURDENED + RNT_BURDENED,
    TOTAL_BURDEN_DENOM = OWN_BURDEN_TOTE + RNT_BURDEN_TOTE,
    BURDEN_RATE    = (TOTAL_BURDENED / TOTAL_BURDEN_DENOM) * 100,

    # SNAP receipt rate: households receiving SNAP / total households
    SNAP_RATE      = (SNAP_RCVDE   / SNAP_TOTE)        * 100
  ) %>%

)

# Step 1 - prepare block group residuals for spatial analysis
# drop NAs because spdep can't handle missing values
bg_residuals_clean <- block_group_residuals %>%
  filter(!is.na(mean_residual))

# Step 2 - define neighbors using contiguity (shared borders)
# using queen contiguity because we have polygons not points
bg_sp <- as(bg_residuals_clean, "Spatial")
neighbors <- poly2nb(bg_sp, queen = TRUE)
weights <- nb2listw(neighbors, style = "W", zero.policy = TRUE)
# style="W" = row standardized, each neighbor gets equal weight
# zero.policy=TRUE = allows block groups with no neighbors

# Step 3 - calculate spatial lag of residuals
# spatial lag = average residual of neighboring block groups
bg_residuals_clean$residual_lag <- lag.listw(weights, 
                                              bg_residuals_clean$mean_residual,
                                              zero.policy = TRUE)
